22def get_customers(route, date=None): 

23 """ 

24 :param route: route instance 

25 :param date: datetime or None 

26 get the customers in the correct order on the route 

27 if date is passed, filter out the ones that don't receive meals 

28 query google maps api to add data to the objects then return the list 

29 """ 

30 # get_customer_order returns a list of pks, we need customer objects 

31 if date: 

32 date = datetime.date(year=date.year, month=date.month, day=date.day) 

33 customers = Customer.objects.filter(route=route).order_by("_order") 

34 if date is not None: 

35 customers = [c for c in customers if c.num_meals_on_day(date)] 

36 return customers
