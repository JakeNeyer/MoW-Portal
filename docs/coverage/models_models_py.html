164class Assignment(models.Model): 

165 """ 

166 Recurring Assignments 

167 day_of_month (date recurrence eg. first monday, second Tuesday) 

168 volunteer (FK) 

169 job (FK) 

170 

171 HOW OPEN ROUTE WORKS: 

172 Before, "Open Route" was a magic volunteer user used to allow substitutions on 

173 days with no recurring assignments. 

174 

175 Now, a NULL volunteer slot in the recurrences table means that this route needs 

176 to be substitued - an OPEN ROUTE. 

177 If there is NO entry for a given day on the month in this table, the route simply 

178 does not run on this day. 

179 """ 

180 

181 volunteer = models.ForeignKey(Volunteer, on_delete=models.SET_NULL, null=True, blank=True) 

182 job = models.ForeignKey(Job, on_delete=models.CASCADE, null=False, blank=False)

188 blank=False, choices=((x, weeks_of_month[x]) for x in range(1, 6))

191 @staticmethod 

192 def actuals( 

193 start_date, 

194 end_date=None, 

195 *, 

196 volunteer=NO_FILTER_SENTINAL, 

197 original=NO_FILTER_SENTINAL, 

198 job=NO_FILTER_SENTINAL, 

199 order_by=None, 

200 exclude_unfilled=False, 

201 ): 

202 """ 

203 This static functions should be called on "Assignment", similar to how 

204 .objects is accessed on Assignment. However, unlike .objects, this does not 

205 actually return a queryset. Instead, this returns a lazily evaluated iterable 

206 of namedtuples, Actuals. Actuals represent what is actually happenning with a job 

207 on a day, including who is volunteering for that job and who the orignal volunteer 

208 is. This method respects unsubstituted assignments, substituted assignments, and 

209 historical data. 

210 

211 Because this function does not return a queryset, it cannot be filtered like one. 

212 To filter the results, filters must be passed into the call to Assignment.actuals(). 

213 For example, a date must be suplied. If only one date is supplied, the actuals are 

214 computed for that date only. If another date is supplied, the actuals for the dates 

215 between start_date and end_date are computed. end_date is exclusive. 

216 Additional filters can be provided, volunteer, original, and job. Note that, right now, 

217 only equality filters are supported. For example, a job__job_type= filter is not 

218 currently possible. 

219 

220 A note on filtering by volunteer=None and original=None: cases where this is true: 

221 - Open Route Assignments 

222 - Unfilled Substitution requests for Open Route Assignments 

223 - Volunteer Records where a Volunteer has since been deleted 

224 This is why there is an additional field, is_substitution. In the case of an Open Route 

225 Assignment, is_substitution will be False. If the actual is an unfilled substitution  

226 request for an Open Route, is_substitution will be True. If the actual is a record, but 

227 a volunteer has been set to None, is_substitution will be whatever it started as. 

228 Also note that in the case of an unfilled substitution request for an open job, only 

229 the substitution request will be returned. 

230 

231 Note: Please do not try to implement this logic yourself. This function has been heavily 

232 tested, and it's important that the whole app is consistent in how actuals are computed. 

233 """ 

234 if order_by: 

235 if isinstance(order_by, str): 

236 order_by = (order_by,) 

237 else: 

238 order_by = () 

239 allowed_order_bys = ("volunteer", "job", "date") 

240 for o in order_by: 

241 if o not in allowed_order_bys: 

242 raise ValueError( 

243 f"""Unsupported order_by, {o}. 

244Can only order by {", ".join(allowed_order_bys)}""" 

245 ) 

246 

247 today = date.today() 

248 end_date = end_date or start_date + timedelta(days=1) 

249 start_date = date(year=start_date.year, month=start_date.month, day=start_date.day) 

250 end_date = date(year=end_date.year, month=end_date.month, day=end_date.day) 

251 

252 assignment_filters = Q() 

253 substitute_filters = Q() 

254 record_filters = Q() 

255 if volunteer != NO_FILTER_SENTINAL: 

256 assignment_filters &= Q(volunteer=volunteer) 

257 substitute_filters &= Q(volunteer=volunteer) 

258 record_filters &= Q(volunteer=volunteer) 

259 if original != NO_FILTER_SENTINAL: 

260 assignment_filters &= Q(volunteer=original) 

261 substitute_filters &= Q(assignment__volunteer=original) 

262 record_filters &= Q(original=original) 

263 if job != NO_FILTER_SENTINAL: 

264 assignment_filters &= Q(job=job) 

265 substitute_filters &= Q(assignment__job=job) 

266 record_filters &= Q(job=job) 

267 

268 exclusor = {"volunteer": None} if exclude_unfilled else {} 

269 

270 assignment_values = dict( 

271 # actual_date=Value(day, output_field=models.DateField()), handled in loop 

272 actual_volunteer=F("volunteer"), 

273 actual_job=F("job"), 

274 actual_original=F("volunteer"), 

275 actual_is_substitution=Value(False, models.BooleanField()), 

276 ) 

277 substitute_values = dict( 

278 actual_date=F("date"), 

279 actual_volunteer=F("volunteer"), 

280 actual_job=F("assignment__job"), 

281 actual_original=F("assignment__volunteer"), 

282 actual_is_substitution=Value(True, models.BooleanField()), 

283 ) 

284 record_values = dict( 

285 actual_date=F("date"), 

286 actual_volunteer=F("volunteer"), 

287 actual_job=F("job"), 

288 actual_original=F("original"), 

289 actual_is_substitution=F("is_substitution"), 

290 ) 

291 if "volunteer" in order_by: 

292 for vals in (assignment_values, substitute_values, record_values): 

293 vals.update( 

294 volunteer_last_name=F("volunteer__user__last_name"), 

295 volunteer_first_name=F("volunteer__user__first_name"), 

296 ) 

297 if "job" in order_by: 

298 for vals in (assignment_values, record_values): 

299 vals.update( 

300 job_type=F("job__job_type__name"), 

301 route_number=F("job__route__number"), 

302 name=F("job__name"), 

303 ) 

304 substitute_values.update( 

305 job_type=F("assignment__job__job_type__name"), 

306 route_number=F("assignment__job__route__number"), 

307 name=F("assignment__job__name"), 

308 ) 

309 

310 assignments = [] 

311 day = max(start_date, today) 

312 while day < end_date: 

313 dom = date_to_day_of_month(day) 

314 exclusionary_subs = Substitution.objects.filter(date=day).values_list( 

315 "assignment_id" 

316 ) 

317 days_assignments = ( 

318 Assignment.objects.filter( 

319 day_of_week=dom.day_of_week, week_of_month=dom.week_of_month 

320 ) 

321 .filter(assignment_filters) 

322 .exclude(id__in=exclusionary_subs) 

323 .exclude(**exclusor) 

324 .values( 

325 actual_date=Value(day, output_field=models.DateField()), 

326 **assignment_values, 

327 ) 

328 ) 

329 assignments.append(days_assignments) 

330 day += timedelta(days=1) 

331 

332 subs = ( 

333 Substitution.objects.filter(date__gte=max(start_date, today), date__lt=end_date) 

334 .filter(substitute_filters) 

335 .exclude(**exclusor) 

336 .values(**substitute_values) 

337 ) 

338 past = ( 

339 VolunteerRecord.objects.filter(date__gte=start_date, date__lt=date.today()) 

340 .filter(record_filters) 

341 .values(**record_values) 

342 ) # Don't exclude Nones, they are deleted vols 

343 result = past.union(subs, *assignments) 

344 

345 new_order_by = [] 

346 for o in order_by: 

347 if o == "volunteer": 

348 new_order_by.append("volunteer_last_name") 

349 new_order_by.append("volunteer_first_name") 

350 elif o == "date": 

351 new_order_by.append("actual_date") 

352 elif o == "job": 

353 new_order_by.append("route_number") 

354 new_order_by.append("job_type") 

355 new_order_by.append("name") 

356 

357 result = result.order_by(*new_order_by) 

358 return map(actual_dict_to_namedtuple, result)

543 def num_meals_on_day(self, day): 

544 """ 

545 Return the number of meals on a specific day 

546 """ 

547 day = date(year=day.year, month=day.month, day=day.day) 

548 if is_weekend(day): 

549 return 0 

550 

551 if day < date.today(): # Strict less-than VERY import here. Otherwise recursion! 

552 try: 

553 record = CustomerRecord.objects.get(date=day, customer=self) 

554 except CustomerRecord.DoesNotExist: 

555 return 0 

556 return record.num_meals 

557 

558 if not self.active: 

559 return False 

560 dt = datetime(year=day.year, month=day.month, day=day.day, hour=0, minute=0, second=0) 

561 possible_match = self.meal_recurrence.between( 

562 dt, dt, dtstart=dt - timedelta(days=1), dtend=dt + timedelta(days=1), inc=True, 

563 ) 

564 drs = DateRange.objects.filter(customer=self, start_date__lte=dt, end_date__gte=dt) 

565 if len(possible_match) == 0 or drs: 

566 return 0 

567 

568 return 1 + (self.num_weekend_meals if is_friday(day) else 0)
