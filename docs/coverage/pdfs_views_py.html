71def substitutions_report(request, begin_date, end_date): 

72 """ 

73 gets all of the information required for a substitution report 

74 takes in a beginning date and end date, displays for these days and 

75 everything in between 

76 """ 

77 day_dict = {} # passed into context 

78 begin_date_datetime = datetime.datetime.strptime(begin_date, "%m-%d-%Y") 

79 end_date_datetime = datetime.datetime.strptime(end_date, "%m-%d-%Y") 

80 count = (end_date_datetime - begin_date_datetime).days 

81 # loop over the count 

82 for i in range(count + 1): 

83 day = begin_date_datetime + datetime.timedelta(days=i) 

84 day_dict[day] = [] 

85 # ge the subs table, loop over and create the person objects 

86 subs_on_day = Substitution.objects.filter(date=day) 

87 # loop over the substitutions for this day 

88 for sub in subs_on_day: 

89 # create display item for context 

90 list_item = SubDisplay() 

91 list_item.job_name = sub.assignment.job.name 

92 list_item.route_number = sub.assignment.job.get_route_number() 

93 if sub.volunteer: 

94 list_item.volunteer = ( 

95 sub.volunteer.user.first_name + " " + sub.volunteer.user.last_name 

96 ) 

97 else: 

98 # it is open if there's not volunteer 

99 list_item.volunteer = "Open Substitution Request" 

100 if sub.assignment.volunteer: 

101 list_item.o_assignment = ( 

102 sub.assignment.volunteer.user.first_name 

103 + " " 

104 + sub.assignment.volunteer.user.last_name 

105 ) 

106 else: 

107 # it is open if there's not volunteer 

108 list_item.o_assignment = OPEN_ROUTE 

109 day_dict[day].append(list_item) 

110 

111 day_dict[day] = list(sorted(day_dict[day], key=lambda x: (x.job_name, x.route_number))) 

112 

113 template = get_template("pdfs/substitutions-report.html") 

114 return to_pdf( 

115 template.render( 

116 {

149def monthly_billing_report(request, begin_date, end_date): 

150 

151 # Parse some dates 

152 begin_date = datetime.datetime.strptime(begin_date, "%m-%d-%Y") 

153 end_date = datetime.datetime.strptime(end_date, "%m-%d-%Y") 

154 

155 # Fetch all the CustomerRecord objects within the given range and sum up the total meals per day 

156 date_meals_query = ( 

157 CustomerRecord.objects.filter(date__gte=begin_date, date__lte=end_date) 

158 .values("date") 

159 .annotate(total_meals=Sum("num_meals")) 

160 .order_by("date") 

161 ) 

162 

163 # Convert the QuerySet to a usable dictionary 

164 date_meals = {} 

165 total = 0 

166 for record in date_meals_query: 

167 key = record["date"] 

168 value = record["total_meals"] 

169 

170 date_meals[key] = value 

171 total += value 

172 

173 # Fetch all the customers within query and count their number of meals 

174 customer_meals_query = ( 

175 CustomerRecord.objects.filter(date__gte=begin_date, date__lte=end_date) 

176 .values("customer", "payment_type", "route_assigned") 

177 .annotate(total_meals=Sum("num_meals")) 

178 .order_by("payment_type") 

179 ) 

180 

181 # Front-end logic handles separation and rendering of customer_meals QuerySet 

182 

183 # Last thing we need is ability to count meals by payment type 

184 payment_meals_query = ( 

185 CustomerRecord.objects.filter(date__gte=begin_date, date__lte=end_date) 

186 .values("payment_type") 

187 .annotate(total_meals=Sum("num_meals"), num_customers=Count("customer", distinct=True)) 

188 .order_by("payment_type") 

189 ) 

190 

191 # Convert to usable dictionary 

192 payments_meals = {} 

193 payments_customers = {} 

194 for elem in payment_meals_query: 

195 payment, meals, customers = ( 

196 elem["payment_type"], 

197 elem["total_meals"], 

198 elem["num_customers"], 

199 ) 

200 

201 payments_meals[payment] = meals 

202 payments_customers[payment] = customers 

203 

204 template = get_template("pdfs/monthly-billing.html") 

205 

206 return to_pdf( 

207 template.render( 

208 { 

209 "date_meals": date_meals, 

210 "total": total, 

211 "cust_meals": customer_meals_query, 

212 "pay_meals": payments_meals, 

213 "pay_customers": payments_customers, 

214 "today": datetime.datetime.now(), 

215 "begin_date": begin_date, 

216 "end_date": end_date, 

217 } 

218 )
