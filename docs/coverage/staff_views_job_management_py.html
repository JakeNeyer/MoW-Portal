57def populate_structs(date_list, date, job): 

58 if not date_list or date != date_list[-1].date: 

59 date_list.append(Date(date)) 

60 job_type = job.job_type 

61 job_types = date_list[-1].job_types 

62 if not job_types or job_type != job_types[-1].job_type: 

63 job_types.append(JobTypeStruct(job_type)) 

64 jobs = job_types[-1].jobs 

65 if not jobs or job != jobs[-1].job: 

66 jobs.append(JobStruct(job, route_number=job.get_route_number())) 

67 return jobs[-1]

105def actuals_to_display( 

106 start_date, 

107 end_date=None, 

108 *, 

109 volunteer=NO_FILTER_SENTINAL, 

110 original=NO_FILTER_SENTINAL, 

111 job=NO_FILTER_SENTINAL, 

112 exclude_unfilled=False, 

113 **kwargs, # support future changes to actuals() interface 

114): 

115 actuals = Assignment.actuals( 

116 start_date, 

117 end_date, 

118 volunteer=volunteer, 

119 original=original, 

120 job=job, 

121 order_by=("date", "job", "volunteer"), # must be totally sorted 

122 exclude_unfilled=exclude_unfilled, 

123 **kwargs, 

124 ) 

125 actuals = list(actuals) 

126 all_jobs = list(Job.objects.all().order_by("route__number", "job_type", "name")) 

127 jobs_index = 0 

128 

129 dates = [] 

130 for i, a in enumerate(actuals): 

131 

132 # insert unassigned jobs from last day 

133 while jobs_index < len(all_jobs) and a.job != all_jobs[jobs_index]: 

134 job_struct = populate_structs( 

135 dates, actuals[max(i - 1, 0)].date, all_jobs[jobs_index] 

136 ) 

137 jobs_index += 1 

138 

139 # insert unassigned jobs from today 

140 if jobs_index == len(all_jobs): 

141 jobs_index = 0 

142 while a.job != all_jobs[jobs_index]: 

143 job_struct = populate_structs(dates, a.date, all_jobs[jobs_index]) 

144 jobs_index += 1 

145 

146 # insert this job and get the volunteer list 

147 todays_volunteers = populate_structs(dates, a.date, a.job).todays_volunteers 

148 

149 # add this volunteer's information 

150 todays_volunteers.append(actual_to_person(a)) 

151 if a.volunteer and a.volunteer.user.email: 

152 dates[-1].job_types[-1].jobs[-1].emails.add(a.volunteer.user.email) 

153 dates[-1].job_types[-1].emails.add(a.volunteer.user.email) 

154 dates[-1].emails.add(a.volunteer.user.email) 

155 

156 # insert unfilled jobs to the end 

157 if "a" in locals(): 

158 for ji in range(jobs_index, len(all_jobs)): 

159 job_struct = populate_structs(dates, a.date, all_jobs[ji]) 

160 

161 # comma-seperate emails 

162 for d in dates: 

163 d.email_str = ",".join(d.emails) 

164 for jt in d.job_types: 

165 jt.email_str = ",".join(jt.emails) 

166 for j in jt.jobs: 

167 j.email_str = ",".join(j.emails) 

168 return dates

374def get_emails_by_job(job_name): 

375 """ 

376 get the list of emails for everyone who has 

377 any sort of assignment for a job 

378 return assigned_emails, sub_emails, and all_emails 

379 """ 

380 # get job 

381 job = Job.objects.get(name=job_name) 

382 # get the emails of anyone ever assigned 

383 assignments = Assignment.objects.filter(job=job) 

384 assigned_emails = [] 

385 for assignment in assignments: 

386 if assignment.volunteer: 

387 email = assignment.volunteer.user.email 

388 if email: 

389 assigned_emails.append(email) 

390 

391 # get the emails of anyone who may be substituting 

392 substitutions = Substitution.objects.filter( 

393 assignment__job=job, date__gte=datetime.date.today() 

394 ) 

395 sub_emails = [] 

396 for sub in substitutions: 

397 # get the email, if its not in the ones we 

398 # got in the assignments, append it 

399 if sub.volunteer: 

400 email = sub.volunteer.user.email 

401 if email: 

402 sub_emails.append(email) 

403 

404 # join with comma 

405 # return all of the emails 

406 return Emails(assigned_emails, sub_emails)
